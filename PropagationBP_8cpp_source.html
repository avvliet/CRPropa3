19  PropagationBP::Y PropagationBP::dY(Vector3d pos, Vector3d dir, double step,
20  double z, double q, double m) const {
21  // half leap frog step in the position
22  pos += dir * step / 2.;
23 
24  // get B field at particle position
25  Vector3d B = getFieldAtPosition(pos, z);
26 
27  // Boris help vectors
28  Vector3d t = B * q / 2 / m * step / c_light;
29  Vector3d s = t * 2 / (1 + t.dot(t));
30  Vector3d v_help;
31 
32  // Boris push
33  v_help = dir + dir.cross(t);
34  dir = dir + v_help.cross(s);
35 
36  // the other half leap frog step in the position
37  pos += dir * step / 2.;
38  return Y(pos, dir);
39  }

62  void PropagationBP::process(Candidate *candidate) const {
63  // save the new previous particle state
64  ParticleState &current = candidate->current;
65  candidate->previous = current;
66 
67  // calculate charge of particle
68  double q = current.getCharge();
69 
70  // rectilinear propagation for neutral particles
71  if (q == 0) {
72  double step = clip(candidate->getNextStep(), minStep, maxStep);
73  Vector3d pos = current.getPosition();
74  Vector3d dir = current.getDirection();
75  current.setPosition(pos + dir * step);
76  candidate->setCurrentStep(step);
77  candidate->setNextStep(maxStep);
78  return;
79  }
80 
81  // further particle parameters
82  double z = candidate->getRedshift();
83  double m = current.getEnergy()/(c_light * c_light);
84 
85  // if minStep is the same as maxStep the adaptive algorithm with its error
86  // estimation is not needed and the computation time can be saved:
87  if (minStep == maxStep){
88  double step = minStep;
89  Vector3d pos = current.getPosition();
90  Vector3d dir = current.getDirection();
91  // half leap frog step in the position
92  Y yOut = dY(pos, dir, step, z, q, m);
93 
94  // full leap frog step in the velocity
95  candidate->current.setDirection(yOut.u);
96  candidate->current.setPosition(yOut.x);
97  candidate->setCurrentStep(step);
98  candidate->setNextStep(step);
99  return;
100  }
101 
102  double step = clip(candidate->getNextStep(), minStep, maxStep);
103  double newStep = step;
104  double r = 42; // arbitrary value > 1
105  Y yIn(current.getPosition(), current.getDirection());
106  Y yOut, yErr;
107 
108  // try performing step until the target error (tolerance) or the minimum step size has been reached
109  while (true) {
110  tryStep(yIn, yOut, yErr, step, current, z, m, q);
111  r = yErr.u.getR() / tolerance; // ratio of absolute direction error and tolerance
112  if (r > 1) { // large direction error relative to tolerance, try to decrease step size
113  if (step == minStep) // already minimum step size
114  break;
115  else {
116  newStep = step * 0.95 * pow(r, -0.2);
117  newStep = std::max(newStep, 0.1 * step); // limit step size decrease
118  newStep = std::max(newStep, minStep); // limit step size to minStep
119  step = newStep;
120  }
121  } else { // small direction error relative to tolerance, try to increase step size
122  if (step != maxStep) { // already maximum step size
123  newStep = step * 0.95 * pow(r, -0.2);
124  newStep = std::min(newStep, 5 * step); // limit step size increase
125  newStep = std::min(newStep, maxStep); // limit step size to max Step
126  }
127  break;
128  }
129  }
130 
131  current.setPosition(yOut.x);
132  current.setDirection(yOut.u.getUnitVector());
133  candidate->setCurrentStep(step);
134  candidate->setNextStep(newStep);
135  }

160  // compare the position after one step with the position after two steps with step/2.
